118  // The gist is, to find if we're within a given distance between two vectors you can use:
119  //
120  // bool within3DManhattanDistance(Vec3 c1, Vec3 c2, float distance)
121  // {
122  // float dx = abs(c2.x - c1.x);
123  // if (dx > distance) return false; // too far in x direction
124  //
125  // float dy = abs(c2.y - c1.y);
126  // if (dy > distance) return false; // too far in y direction
127  //
128  // float dz = abs(c2.z - c1.z);
129  // if (dz > distance) return false; // too far in z direction
130  //
131  // return true; // we're within the cube
132  // }
133  //
134  // Or to just calculate the straight Manhattan distance you could use:
135  //
136  // float getManhattanDistance(Vec3 c1, Vec3 c2)
137  // {
138  // float dx = abs(c2.x - c1.x);
139  // float dy = abs(c2.y - c1.y);
140  // float dz = abs(c2.z - c1.z);
141  // return dx+dy+dz;
142  // }
143  //

145  {
146  T dx = v2.x - v1.x;
147  T dy = v2.y - v1.y;
148  T dz = v2.z - v1.z;
